	title = gibraltar.1.t
	desc = gibraltar.1.d
	picture = GFX_report_event_generic_conference
	
	is_triggered_only = yes
	
	option = { #sure
		name = gibraltar.1.a
		ai_chance = {
			factor = 80
			modifier = {
				threat > 0.2
				factor = 5
			}
			modifier = {
				threat > 0.4
				factor = 5
			}
		}
		GIB = { country_event = gibraltar.2 }
		give_guarantee = GIB
		hidden_effect = {
			news_event = { id = liberation_news.1 days = 1 }
		}
	}

	option = { #no
		name = gibraltar.1.b
		ai_chance = {
			factor = 20
		}
		GIB = { country_event = gibraltar.3 }	
	}

}

#Gibralta wants to intervene in civil war -Portugal
country_event = {
	id = gibraltar.4 
	title = gibraltar.4.t
	desc = gibraltar.4.d
	picture = GFX_report_event_poland_rejects
	
	is_triggered_only = yes

	option = { #cool
		name = gibraltar.4.a
		
		every_country = {
			limit = {
				original_tag = SPR
			}
			GIB = {
				declare_war_on = {
                    target = PREV
                    type = annex_everything
				}
			}
			POR = {
				declare_war_on = {
                    target = PREV
                    type = annex_everything
				}
			}
        }
		
	}

}
